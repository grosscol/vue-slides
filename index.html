Anatomy of a Component

<template>
  <!-- HTML goes here -->
</template>
<script>
 // Javascript goes here
</script>

Example

<template>
  <button @click="do_stuff">Gadget {{count}}</button>
</template>
<script>
export default {
  name: "gadgetButton",
  data() {
    return {
      count: 0
    }
  },
  methods: {
    do_stuff: function() {
      this.count++
    }
  }
}
</script>

Provide

Provided values can be acceessed by decendants at and depth

<script>
export default {
  name: 'GeneDashboard',
  provide: function() {
    return {
      chrom: this.chrom,
      start: this.start,
      stop: this.stop,
    }
  }
}
</script>

Inject

Injected data provided by some ancestor

<template>
  <div class="child-component">
    <h6>{{geneData.full_gene_name}}</h6>
      Region: <span>{{chrom}}:{{start}}-{{stop}}</span>
  </div>
</template>
export default {
  name: "GeneInfo",
  inject: {
    chrom: {default: 0},
    start: {default: 0},
    stop: {default: 1},
  }
}

Emit

Events passed to ancestors

<template>
 <!-- Search bar context -->
<AutoComplete v-on:suggestionSelect="doSuggest"></AutoComplete>
</template>
<script>
export default {
  methods:
doSuggest(suggestion){
  let resultTicket = this.suggestToResultTicket(suggestion)

  if(resultTicket.endpoint === "notfound"){
    this.$emit("noSearchResults", resultTicket.query)
  }
}

Vue Lifecycle Hooks

Vue Lifecycle

Exmaple: Registering Events

export default {
  name: SeqDepth,
  mounted: function() {
    window.addEventListener("resize", this.debouncedDraw);
  },
  unmounted: function() {
    window.removeEventListener("resize", this.debouncedDraw);
  }
}
